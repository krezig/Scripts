===Compilar esuqema Completo
EXEC DBMS_UTILITY.COMPILE_SCHEMA('RECAUDADOR');

=== Consultar objetos Invalidos
SELECT OWNER,OBJECT_NAME, OBJECT_TYPE,CREATED,STATUS
FROM ALL_OBJECTS
WHERE STATUS = 'INVALID'
and owner in ('FACTURACION','RECAUDADOR','OBLI_RECAUDADOR','RELOG_RECAUDADOR');


=== Contar # de objetos Invalidos
SELECT count(*) as "Objetos invalidos"
FROM ALL_OBJECTS
WHERE STATUS = 'INVALID'
and owner in ('FACTURACION','RECAUDADOR','OBLI_RECAUDADOR','RELOG_RECAUDADOR');


=== Compilar todos los objetos invidualmente
SELECT OWNER,OBJECT_NAME, OBJECT_TYPE,'ALTER '||OBJECT_TYPE|| ' ' ||OWNER|| '.'||OBJECT_NAME ||' COMPILE;' AS "Query Compilado"
FROM ALL_OBJECTS
WHERE STATUS = 'INVALID' and  object_type != 'PACKAGE BODY' and  object_type not like '%JAVA%'
and owner in ('FACTURACION','RECAUDADOR','OBLI_RECAUDADOR','RELOG_RECAUDADOR')
ORDER  BY 4 DESC;


=== Compilar todos los BODY invidualmente
SELECT OWNER,OBJECT_NAME, OBJECT_TYPE,'ALTER PACKAGE'|| ' ' ||OWNER|| '.'||OBJECT_NAME ||' COMPILE BODY;' AS "Query Compilado"
FROM ALL_OBJECTS
WHERE STATUS = 'INVALID'
and object_type = 'PACKAGE BODY'
and owner in ('FACTURACION','RECAUDADOR','OBLI_RECAUDADOR','RELOG_RECAUDADOR');


=== Eliminiar todas las JAVAS
SELECT status,OWNER,OBJECT_NAME, OBJECT_TYPE,'DROP '||OBJECT_TYPE|| ' ' ||OWNER|| '."'||OBJECT_NAME ||'";' AS "Query Compilado"
FROM ALL_OBJECTS
WHERE OBJECT_TYPE LIKE '%JAVA%'
and owner in ('FACTURACION','RECAUDADOR','OBLI_RECAUDADOR','RELOG_RECAUDADOR')
ORDER BY 3 DESC;
